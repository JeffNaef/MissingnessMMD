# Sample a dataset of size n from univariate Cauchy with a portion of outliers
def datagen(n,theta,contamination = []):
    datag = cauchy.rvs(loc=theta,scale=1,size=n-size(contamination))
    datag = concatenate((datag,contamination),axis=0)
    return(datag)

# Gaussian kernel of parameter gamma=1
# Minibatch size = sample size = n, number of iterations = Nstep = 1000
def MMD(data,Nstep=1000,gamma=1,eta=0.1):
    theta = 0.0
    n = size(data)
    for t in arange(Nstep):
        sample = cauchy.rvs(loc=theta,scale=1,size=n)
        K1 = outerdiff(v1=sample,v2=data)
        K2 = outerdiff(v1=sample,v2=sample)
        diff = 2*(theta*ones(n)-sample)/(1+(theta*ones(n)-sample)**2)
        gradient = -2*(mean(diff*(exp(-gamma*multiply(K2,K2))-diag(ones(n))))/(n-1)-mean(diff*exp(-gamma*multiply(K1,K1))))
        theta = theta-eta*gradient/sqrt(t+1)
    return(theta)

# Outliers sampled from a standard Cauchy
# Sample size n, repeat the experience N times, percentage max of outliers=(outl-1)*5
start_time = time.time()
theta = 2
n = 200
N = 20
outl = 9
resultats = zeros([3*outl,N])
for n_cont in arange(outl):
    for i in arange(N):
        X = datagen(n=n,theta=theta,contamination=2*random.standard_cauchy(n_cont*5))
        resultats[n_cont*3,i] = (theta-MMD(X))**2
        resultats[n_cont*3+1,i] = (theta-MLE(X))**2
        resultats[n_cont*3+2,i] = (theta-GEO_MED(X))**2
        #resultats[n_cont*4+3,i] = (theta-MMD_VI(X))**2
    vec_res = sqrt(mean(resultats,axis=1))
end_time = time.time()
print('Time :', end_time-start_time, 'seconds')
for n_cont in arange(outl):
    print('SqMSE of MMD for', n_cont*2.5, '% of outliers :', (vec_res[n_cont*3+0]))
    print('SqMSE of MLE for', n_cont*2.5, '% of outliers :', (vec_res[n_cont*3+1]))
    print('SqMSE of MED for', n_cont*2.5, '% of outliers :', (vec_res[n_cont*3+2]))
    #print('SqMSE of VI for', n_cont*2.5, '% of outliers :', (vec_res[n_cont*4+3]))
